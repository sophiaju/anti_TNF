model_performance_data = {}
# store random forest feature importances
importance_snp = {}

for model in models:
    model_name = model_abbrev[type(model).__name__]
    # fit model on training data
    model.fit(X_train, Y_train)
    # predict on both training and testing
    train_pred = model.predict(X_train)
    test_pred = model.predict(X_test)
    
    # pos_label and auroc are inverted bc model is predicting severity of disease, i.e. neg or 0 DAS is responder
    fpr, tpr, thresholds = roc_curve(Y_test_bin, test_pred, pos_label=0)
